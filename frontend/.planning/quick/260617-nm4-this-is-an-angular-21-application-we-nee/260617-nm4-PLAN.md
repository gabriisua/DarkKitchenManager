<objective>
Separate Menu download functionalities (PDF, labels) from editing into a dedicated Menu Detail view.

**Purpose:** The MenuFormComponent is currently overloaded — it handles both editing/creating menus AND downloading PDFs/labels. By extracting download actions into a standalone MenuDetailComponent, the form becomes focused on editing only, and users get a clean read-only detail view with all download CTA buttons.

**Output:**
- Cleaned MenuFormComponent (no download methods/buttons)
- Updated MenuPageComponent (detail navigation from list)
- New MenuDetailComponent (read-only detail with download CTAs)
- Updated routing (/menus/:id)
</objective>

    **1. Create `src/app/features/dashboard/menu/menu-detail/menu-detail.component.ts`:**

    Create a standalone Angular component following the pattern established by `PlateDetailComponent`:

    ```typescript
    import { Component, OnInit, inject, signal } from '@angular/core';
    import { CommonModule } from '@angular/common';
    import { ActivatedRoute, Router, RouterModule } from '@angular/router';
    import { MenuService } from '../../../../core/services/menu.service';
    import { UiService } from '../../../../core/services/ui.service';
    import { MenuDetail } from '../../../../shared/models/api.models';

    import { MatCardModule } from '@angular/material/card';
    import { MatButtonModule } from '@angular/material/button';
    import { MatIconModule } from '@angular/material/icon';
    import { MatTooltipModule } from '@angular/material/tooltip';
    import { MatTableModule } from '@angular/material/table';
    import { MatDividerModule } from '@angular/material/divider';

    @Component({
      selector: 'app-menu-detail',
      standalone: true,
      imports: [
        CommonModule, RouterModule,
        MatCardModule, MatButtonModule, MatIconModule,
        MatTooltipModule, MatTableModule, MatDividerModule
      ],
      templateUrl: './menu-detail.component.html',
      styleUrls: ['./menu-detail.component.scss']
    })
    export class MenuDetailComponent implements OnInit {
      private route = inject(ActivatedRoute);
      private router = inject(Router);
      private menuService = inject(MenuService);
      private uiService = inject(UiService);

      menu = signal<MenuDetail | null>(null);
      loading = signal(true);

      displayedColumns = ['plateName', 'actions'];

      ngOnInit(): void {
        const id = Number(this.route.snapshot.paramMap.get('id'));

        if (!id) {
          this.uiService.showToast('ID Menu non valido', 'error');
          this.router.navigate(['/menus']);
          return;
        }

        this.loadMenu(id);
      }

      private loadMenu(id: number): void {
        this.uiService.showLoader('Caricamento menu...');
        this.loading.set(true);

        this.menuService.getById(id).subscribe({
          next: (res) => {
            this.menu.set(res.data || res);
            this.loading.set(false);
            this.uiService.hideLoader();
          },
          error: () => {
            this.uiService.hideLoader();
            this.loading.set(false);
            this.uiService.showToast('Errore nel caricamento del menu', 'error');
            this.router.navigate(['/menus']);
          }
        });
      }

      downloadMenuPdf(): void {
        const currentMenu = this.menu();
        if (!currentMenu) return;

        this.uiService.showLoader('Generazione PDF menu...');
        this.menuService.downloadMenuPdf(currentMenu.id).subscribe({
          next: (blob) => {
            this.uiService.hideLoader();

            const url = window.URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `menu-${currentMenu.id}.pdf`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            window.URL.revokeObjectURL(url);
          },
          error: () => {
            this.uiService.hideLoader();
            this.uiService.showToast('Errore durante il download del PDF', 'error');
          }
        });
      }

      downloadClassicLabel(plateId: number): void {
        const currentMenu = this.menu();
        if (!currentMenu) return;

        this.menuService.downloadClassicLabel(currentMenu.id, plateId).subscribe({
          next: (blob) => {
            const url = window.URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `label-classic-${plateId}.pdf`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            window.URL.revokeObjectURL(url);
          },
          error: () => this.uiService.showToast("Errore durante il download dell'etichetta classica", 'error')
        });
      }

      downloadCustomLabel(plateId: number): void {
        const currentMenu = this.menu();
        if (!currentMenu) return;

        this.menuService.downloadCustomLabel(currentMenu.id, plateId).subscribe({
          next: (blob) => {
            const url = window.URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `label-custom-${plateId}.pdf`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            window.URL.revokeObjectURL(url);
          },
          error: () => this.uiService.showToast("Errore durante il download dell'etichetta personalizzata", 'error')
        });
      }

      goBack(): void {
        this.router.navigate(['/menus']);
      }
    }
    ```
