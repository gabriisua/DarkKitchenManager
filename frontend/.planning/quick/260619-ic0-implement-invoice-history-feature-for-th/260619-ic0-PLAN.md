Use `buildPagedParams` from `../../core/utils/http-params.util` for the history list (import it if not already imported — it's not currently; add the import alongside existing HttpClient imports).

```typescript
getInvoiceHistory(params: InvoiceHistoryRequest): Observable<PagedResponse<InvoiceHistoryItem>> {
  return this.http.get<any>(`${this.baseUrl}/history`, { params: buildPagedParams(params) }).pipe(
    map((res: any) => {
      // Handle both wrapped (res.data) and unwrapped responses
      if (res && res.data && res.data.items) {
        return res.data as PagedResponse<InvoiceHistoryItem>;
      }
      if (res && res.items) {
        return res as PagedResponse<InvoiceHistoryItem>;
      }
      return { items: res || [], totalCount: 0, page: params.page, pageSize: params.pageSize, totalPages: 0 } as PagedResponse<InvoiceHistoryItem>;
    })
  );
}

deleteInvoice(id: string): Observable<void> {
  return this.http.delete<void>(`${this.baseUrl}/${id}`);
}

getInvoicePdfUrl(id: string): Observable<string> {
  // Expects response like { pdfDownloadUrl: string } and returns the URL
  return this.http.get<{ pdfDownloadUrl: string }>(`${this.baseUrl}/${id}/pdf-url`).pipe(
    map(res => res.pdfDownloadUrl)
  );
}
```
