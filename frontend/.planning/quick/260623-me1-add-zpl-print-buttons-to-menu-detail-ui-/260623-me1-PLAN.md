       ```typescript
       openSinglePrintDialog(plateId: number): void {
         const dialogRef = this.dialog.open(PrintSettingsDialogComponent, {
           width: '400px'
         });

         dialogRef.afterClosed().subscribe(result => {
           if (!result) return;

           this.uiService.showLoader('Invio stampa ZPL...');
           this.plateService.printSingleLabel(plateId, result).subscribe({
             next: () => {
               this.uiService.hideLoader();
               this.uiService.showToast('Stampa ZPL inviata con successo', 'success');
             },
             error: () => {
               this.uiService.hideLoader();
               this.uiService.showToast('Errore durante l\'invio della stampa ZPL', 'error');
             }
           });
         });
       }

       openBatchPrintDialog(): void {
         const currentMenu = this.menu();
         if (!currentMenu?.menuItems?.length) return;

         const dialogRef = this.dialog.open(PrintSettingsDialogComponent, {
           width: '400px'
         });

         dialogRef.afterClosed().subscribe(result => {
           if (!result) return;

           const batchItems: PrintBatchItem[] = currentMenu.menuItems.map(item => ({
             plateId: item.plateId,
             copies: result.copies,
             pauseAfter: result.pauseAfter,
             lotNumber: result.lotNumber
           }));

           this.uiService.showLoader('Invio stampa ZPL batch...');
           this.plateService.printBatchLabels(batchItems).subscribe({
             next: () => {
               this.uiService.hideLoader();
               this.uiService.showToast('Stampa ZPL batch inviata con successo', 'success');
             },
             error: () => {
               this.uiService.hideLoader();
               this.uiService.showToast('Errore durante l\'invio della stampa ZPL batch', 'error');
             }
           });
         });
       }
       ```
